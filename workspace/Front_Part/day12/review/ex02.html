------- const path = require("path"); // npm i axios // # in terminal const axios = require('axios'); const fastify = require("fastify")({ logger: true, // 로깅을 활성화 }); // npm i @fastify/cors@7.0.0 // fastify-cors 플러그인 추가 fastify.register(require('@fastify/cors'), { origin: "*", // 모든 도메인에서 요청을 허용 }); fastify.get("/", async function (request, reply) { // 환경 변수가 제대로 로드되었는지 확인 const { GEMINI_KEY } = process.env; if (!GEMINI_KEY) { return reply.status(500).send({ error: "GEMINI_KEY 환경 변수가 설정되지 않았습니다." }); } console.log("GEMINI_KEY:", GEMINI_KEY); const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${GEMINI_KEY}`; try { const response = await axios.post(url, { contents: [ { parts: [ { text: "어떡하면 학생들이 저한테 집중해줄까요? 20자 이내로. 핵심만.", }, ], }, ], }); // JSON 형태로 응답이 오기 때문에, 별도로 JSON.parse를 할 필요 없이 바로 사용 가능 const json = response.data; // 응답의 구조에 맞게 필요한 값을 추출 const answer = json.candidates[0].content.parts[0].text; return answer; // 결과를 클라이언트에게 반환 } catch (error) { console.error("Error during API call:", error); return reply.status(500).send({ error: "API 호출 중 오류가 발생했습니다." }); } }); fastify.listen( { port: process.env.PORT || 3000, host: "0.0.0.0" }, function (err, address) { if (err) { console.error(err); process.exit(1); } console.log(`Your app is listening on ${address}`); } );
